    	return new Promise((resolve, reject) => {
    		function next() {
    			const args = Array.isArray(paramQueue[index++]) ? paramQueue[index++] : [paramQueue[index++]];
    			if (index <= paramQueue.length) {
    				Promise.resolve().then(() => fun(...args)).then(next).catch(reject)
    			}
    			else resolve()
    		}

    	})
